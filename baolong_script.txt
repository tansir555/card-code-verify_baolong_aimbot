-- ************************ 仅加密核心的 GitHub 链接 ************************
local function decryptStr(encryptedStr)
    local key = 666999
    local decrypted = ""
    for i = 1, #encryptedStr do
        local charCode = bit32.bxor(string.byte(encryptedStr, i), key) % 256
        decrypted = decrypted .. string.char(charCode)
    end
    return decrypted
end
